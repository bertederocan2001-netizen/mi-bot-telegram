def log_admin(admin_id, accion, detalle):
    cursor.execute(
        "INSERT INTO logs (admin_id, accion, detalle) VALUES (?, ?, ?)",
        (admin_id, accion, detalle)
    )

def crear_usuario(usuario):
    username = usuario.username if usuario.username else "Sin username"

    cursor.execute(
        "SELECT user_id FROM usuarios WHERE user_id=?",
        (usuario.id,)
    )

    existe = cursor.fetchone()

    if not existe:
        cursor.execute(
            "INSERT INTO usuarios (user_id, nombre, username) VALUES (?, ?, ?)",
            (usuario.id, usuario.first_name, username)
        )
    else:
        cursor.execute(
            "UPDATE usuarios SET nombre=?, username=? WHERE user_id=?",
            (usuario.first_name, username, usuario.id)
        )

    db.commit()


def get_moneda(user_id):
    cursor.execute(
        "SELECT moneda_actual FROM usuarios WHERE user_id=?",
        (user_id,)
    )

    r = cursor.fetchone()

    if r:
        return r[0]

    return "TRX"


def set_moneda(user_id, moneda):
    cursor.execute(
        "UPDATE usuarios SET moneda_actual=? WHERE user_id=?",
        (moneda, user_id)
    )

    db.commit()


def get_saldo(user_id, moneda):
    moneda = moneda.lower()

    cursor.execute(
        f"SELECT {moneda} FROM usuarios WHERE user_id=?",
        (user_id,)
    )

    r = cursor.fetchone()

    if r:
        return float(r[0])

    return 0.0


def add_saldo(user_id, moneda, monto):
    moneda = moneda.lower()

    saldo_actual = get_saldo(user_id, moneda)
    nuevo = saldo_actual + float(monto)

    cursor.execute(
        f"UPDATE usuarios SET {moneda}=? WHERE user_id=?",
        (nuevo, user_id)
    )

    db.commit()

    return nuevo


def quitar_saldo(user_id, moneda, monto):
    moneda = moneda.lower()

    saldo_actual = get_saldo(user_id, moneda)

    if saldo_actual < monto:
        return None

    nuevo = saldo_actual - float(monto)

    cursor.execute(
        f"UPDATE usuarios SET {moneda}=? WHERE user_id=?",
        (nuevo, user_id)
    )

    db.commit()

    return nuevo


def guardar_historial(user_id, tipo, moneda, monto):
    cursor.execute(
        "INSERT INTO historial (user_id, tipo, moneda, monto) VALUES (?, ?, ?, ?)",
        (user_id, tipo, moneda, monto)
    )

    db.commit()


def tiene_deposito_pendiente(user_id):
    cursor.execute(
        "SELECT id FROM depositos WHERE user_id=? AND estado='pendiente'",
        (user_id,)
    )

    return cursor.fetchone() is not None


def tiene_retiro_pendiente(user_id):
    cursor.execute(
        "SELECT id FROM retiros WHERE user_id=? AND estado='pendiente'",
        (user_id,)
    )

    return cursor.fetchone() is not None


def es_primer_deposito(user_id):
    cursor.execute(
        "SELECT COUNT(*) FROM depositos WHERE user_id=? AND estado='aprobado'",
        (user_id,)
    )

    cantidad = cursor.fetchone()[0]

    return cantidad == 0

async def recibir_texto(update: Update, context: ContextTypes.DEFAULT_TYPE):

    crear_usuario(update.effective_user)

    texto = update.message.text.strip()
    user_id = update.effective_user.id

    if context.user_data.get("esperando_numero"):

        if len(texto) != 8 or not texto.isdigit():

            await update.message.reply_text(
                "❌ Número inválido\n\n"
                "📌 Debe tener exactamente 8 dígitos.\n"
                "Ejemplo: 55555555"
            )

            return

        context.user_data["numero"] = texto
        context.user_data["esperando_numero"] = False
        context.user_data["esperando_monto"] = True

        try:
            await update.message.delete()
        except:
            pass

        await borrar_anterior(
            update.effective_chat.id,
            context
        )

        mensaje = await context.bot.send_message(
            chat_id=update.effective_chat.id,
            text="💵 Ahora escribe el monto que depositaste:"
        )

        context.chat_data["ultimo_mensaje"] = mensaje.message_id
        return

    if context.user_data.get("esperando_monto"):

        try:
            monto = float(texto.replace(",", "."))
        except:

            await update.message.reply_text(
                "❌ Monto inválido. Ejemplo: 100"
            )

            return

        if monto <= 0:

            await update.message.reply_text(
                "❌ El monto debe ser mayor que 0."
            )

            return

        moneda = get_moneda(user_id)

        context.user_data["monto_deposito"] = monto
        context.user_data["moneda_deposito"] = moneda
        context.user_data["esperando_monto"] = False
        context.user_data["esperando_captura"] = True

        try:
            await update.message.delete()
        except:
            pass

        await borrar_anterior(
            update.effective_chat.id,
            context
        )

        mensaje = await context.bot.send_message(
            chat_id=update.effective_chat.id,
            text=(
                "📸 Ahora manda una captura clara.\n\n"
                f"💰 Monto: {monto} {moneda}\n\n"
                "Debe verse fecha y hora."
            )
        )

        context.chat_data["ultimo_mensaje"] = mensaje.message_id
        return

    if context.user_data.get("esperando_monto_retiro"):

        try:
            monto = float(texto.replace(",", "."))
        except:

            await update.message.reply_text(
                "❌ Monto inválido. Ejemplo: 50"
            )

            return

        moneda = get_moneda(user_id)
        saldo = get_saldo(user_id, moneda)

        if monto <= 0:

            await update.message.reply_text(
                "❌ El monto debe ser mayor que 0."
            )

            return

        if monto > saldo:

            await update.message.reply_text(
                f"❌ Saldo insuficiente.\n\nTienes {saldo} {moneda}"
            )

            return

        context.user_data["monto_retiro"] = monto
        context.user_data["moneda_retiro"] = moneda
        context.user_data["esperando_monto_retiro"] = False
        context.user_data["esperando_cuenta_retiro"] = True

        try:
            await update.message.delete()
        except:
            pass

        await borrar_anterior(
            update.effective_chat.id,
            context
        )

        mensaje = await context.bot.send_message(
            chat_id=update.effective_chat.id,
            text=(
                "🏦 Ahora escribe la cuenta, "
                "tarjeta o wallet donde quieres recibir el retiro."
            )
        )

        context.chat_data["ultimo_mensaje"] = mensaje.message_id
        return

    if context.user_data.get("esperando_cuenta_retiro"):

        cuenta = texto

        monto = context.user_data.get("monto_retiro")
        moneda = context.user_data.get("moneda_retiro")

        cursor.execute("""
        INSERT INTO retiros (user_id, chat_id, moneda, monto, cuenta)
        VALUES (?, ?, ?, ?, ?)
        """, (
            user_id,
            update.effective_chat.id,
            moneda,
            monto,
            cuenta
        ))

        db.commit()

        retiro_id = cursor.lastrowid

        keyboard = [[
            InlineKeyboardButton(
                "✅ Aprobar",
                callback_data=f"aprobar_ret_{retiro_id}"
            ),
            InlineKeyboardButton(
                "❌ Rechazar",
                callback_data=f"rechazar_ret_{retiro_id}"
            )
        ]]

        await context.bot.send_message(
            chat_id=GRUPO_ID,
            text=(
                "💸 NUEVO RETIRO\n\n"
                f"🧾 ID retiro: {retiro_id}\n"
                f"👤 Usuario: {update.effective_user.first_name}\n"
                f"🆔 ID: {user_id}\n\n"
                f"🏆 Moneda: {moneda}\n"
                f"💰 Monto: {monto} {moneda}\n"
                f"🏦 Cuenta/Wallet:\n{cuenta}\n\n"
                "⏳ Estado: pendiente"
            ),
            reply_markup=InlineKeyboardMarkup(keyboard)
        )

        context.user_data["esperando_cuenta_retiro"] = False

        try:
            await update.message.delete()
        except:
            pass

        await borrar_anterior(
            update.effective_chat.id,
            context
        )

        mensaje = await context.bot.send_message(
            chat_id=update.effective_chat.id,
            text=(
                "✅ Tu solicitud de retiro fue enviada.\n\n"
                "🕒 Espera aprobación."
            )
        )

        context.chat_data["ultimo_mensaje"] = mensaje.message_id
        return

async def recibir_captura(update: Update, context: ContextTypes.DEFAULT_TYPE):

    moneda = context.user_data.get("moneda_deposito")

    cursor.execute("""
    INSERT INTO depositos (user_id, chat_id, metodo, numero, moneda, monto, file_id)
    VALUES (?, ?, ?, ?, ?, ?, ?)
    """, (
        usuario.id,
        update.effective_chat.id,
        metodo,
        numero,
        moneda,
        monto,
        foto
    ))

    db.commit()

    deposito_id = cursor.lastrowid

    keyboard = [[

            "✅ Aprobar",

        ),
        InlineKeyboardButton(
            "❌ Rechazar",
            callback_data=f"rechazar_dep_{deposito_id}"
        )
    ]]

    username = (
        f"@{usuario.username}"
        if usuario.username
        else "Sin username"

        pass

        update.effective_chat.id,

    mensaje = await context.bot.send_message(

        text=(
            "✅ Tu depósito fue enviado.\n\n"
            "🕒 Espera aprobación de los administradores."
        )

    except:

        await update.message.reply_text(
            "Uso: /addsaldo ID MONEDA MONTO"
        )

        return

    if moneda not in MONEDAS:

        await update.message.reply_text(

        )
